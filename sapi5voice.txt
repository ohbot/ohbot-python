import asyncio
import winsound

from winrt.windows.media.speechsynthesis import SpeechSynthesizer
from winrt.windows.storage.streams import DataReader


async def speak_to_wav(text, filename="onecore.wav", voice_name=None):
    synth = SpeechSynthesizer()

    voices = list(SpeechSynthesizer.all_voices)

    for v in voices:
        print(v.display_name, v.language)

    if voice_name:
        for v in voices:
            if voice_name.lower() in v.display_name.lower():
                synth.voice = v
                break

    stream = await synth.synthesize_text_to_stream_async(text)

    reader = DataReader(stream.get_input_stream_at(0))
    await reader.load_async(stream.size)

    data = bytearray(stream.size)
    reader.read_bytes(data)

    with open(filename, "wb") as f:
        f.write(data)


asyncio.run(speak_to_wav("Hello from a OneCore voice.", voice_name="Ryan"))
winsound.PlaySound("onecore.wav", winsound.SND_FILENAME)
